## 一、功能概述

Phase 2 实现以下核心功能：

1. **AI 引导式灵感补充** - 调酒师通过对话引导用户完善灵感
2. **多维度完整度评分** - 清晰度、丰富度、可行性、独特性评估
3. **液体动画反馈** - 每完成一轮对话，杯中液体上升

export interface LLMProvider {
  chat(messages: Message[]): Promise<string>
  stream(messages: Message[]): AsyncIterable<string>
}

export class OpenAIProvider implements LLMProvider {
  async chat(messages: Message[]): Promise<string> {
    const response = await fetch('https://api.openai.com/v1/chat/completions', {
      method: 'POST',
      headers: {
        'Content-Type': 'application/json',
        'Authorization': `Bearer ${config.apiKey}`,
      },
      body: JSON.stringify({
        model: config.model,
        messages,
        temperature: config.temperature,
        max_tokens: config.maxTokens,
      }),
    })
    
    const data = await response.json()
    return data.choices[0].message.content
  }
}

#### 3.2.2 评分器

```typescript
// src/services/scoring/evaluator.ts

export class CompletenessEvaluator {
  /**
   * 评估灵感的完整度
   */
  async evaluate(inspiration: Inspiration): Promise<CompletenessScore> {
    const scores = {
      clarity: await this.evaluateClarity(inspiration),
      richness: await this.evaluateRichness(inspiration),
      feasibility: await this.evaluateFeasibility(inspiration),
      uniqueness: await this.evaluateUniqueness(inspiration),
    }
    
    const total = this.calculateTotal(scores)
    
    return {
      ...scores,
      total,
      status: this.getStatus(total),
    }
  }
  
  /**
   * 评估清晰度
   */
  private async evaluateClarity(inspiration: Inspiration): Promise<number> {
    // 检查是否有核心总结
    // 检查是否能用一句话描述
    // 检查边界是否清晰
    const content = inspiration.rawInput.text
    const structured = inspiration.structuredContent
    
    let score = 0
    
    // 有原始内容 +20
    if (content && content.length > 10) score += 20
    
    // 有核心想法 +30
    if (structured?.coreIdea) score += 30
    
    // 有对话记录 +20
    if (inspiration.brewingLog.length > 0) score += 20
    
    // 对话记录超过3轮 +30
    if (inspiration.brewingLog.length >= 3) score += 30
    
    return Math.min(score, 100)
  }
  
  /**
   * 评估丰富度
   */
  private async evaluateRichness(inspiration: Inspiration): Promise<number> {
    // 检查感官元素数量
    // 检查场景描述
    // 检查情感氛围
    const structured = inspiration.structuredContent
    
    let score = 0
    
    // 有感官元素 +25 * 元素数量
    const sensoryCount = structured?.sensoryElements?.length || 0
    score += Math.min(sensoryCount * 25, 50)
    
    // 有情感氛围 +25
    if (structured?.mood) score += 25
    
    // 有关键词 +25
    if (structured?.keywords?.length > 0) score += 25
    
    return Math.min(score, 100)
  }
  
  /**
   * 评估可行性
   */
  private async evaluateFeasibility(inspiration: Inspiration): Promise<number> {
    // 检查是否有落地方向
    const structured = inspiration.structuredContent
    
    let score = 0
    
    // 有可能的方向 +50
    const directions = structured?.possibleDirections?.length || 0
    if (directions > 0) score += 50
    
    // 方向超过2个 +50
    if (directions >= 2) score += 50
    
    return Math.min(score, 100)
  }
  
  /**
   * 评估独特性
   */
  private async evaluateUniqueness(inspiration: Inspiration): Promise<number> {
    // 检查是否有个人视角
    // 检查是否有跨界组合
    const structured = inspiration.structuredContent
    
    let score = 50 // 基础分
    
    // 有个人视角 +25
    if (structured?.personalPerspective) score += 25
    
    // 有跨界组合 +25
    if (structured?.crossDomain) score += 25
    
    return Math.min(score, 100)
  }
  
  /**
   * 计算总分
   */
  private calculateTotal(scores: ScoreDimensions): number {
    return (
      scores.clarity * 0.3 +
      scores.richness * 0.3 +
      scores.feasibility * 0.2 +
      scores.uniqueness * 0.2
    )
  }
  
  /**
   * 获取状态
   */
  private getStatus(total: number): InspirationStatus {
    if (total >= 100) return 'bloom'
    if (total >= 75) return 'bud'
    if (total >= 50) return 'grow'
    if (total >= 25) return 'sprout'
    return 'seed'
  }
}
```

### 3.3 调酒工作台界面

#### 3.3.1 主界面

```typescript
// src/screens/BrewingScreen/index.tsx

import React, { useState, useEffect } from 'react'
import { View, ScrollView, TextInput, TouchableOpacity } from 'react-native'
import { Text, Button } from 'react-native-paper'
import { useNavigation } from '@react-navigation/native'
import { Glass } from '../../components/Glass'
import { ChatBubble } from './ChatBubble'
import { QuestionCard } from './QuestionCard'
import { ProgressIndicator } from './ProgressIndicator'
import { useBrewingStore } from '../../store/brewingStore'
import { useInspirationStore } from '../../store/inspirationStore'

interface BrewingScreenProps {
  route: {
    params: {
      inspirationId: string
    }
  }
}

export const BrewingScreen: React.FC<BrewingScreenProps> = ({ route }) => {
  const { inspirationId } = route.params
  const navigation = useNavigation()
  
  // State
  const [userInput, setUserInput] = useState('')
  const [isTyping, setIsTyping] = useState(false)
  
  // Store
  const inspiration = useInspirationStore(s => s.getInspiration(inspirationId))
  const { 
    messages, 
    currentQuestion,
    addMessage,
    generateNextQuestion,
    isLoading 
  } = useBrewingStore()
  
  // Effects
  useEffect(() => {
    // 初始化对话
    if (messages.length === 0) {
      generateNextQuestion(inspiration)
    }
  }, [])
  
  // Handlers
  const handleSend = async () => {
    if (!userInput.trim()) return
    
    // 添加用户消息
    addMessage({ role: 'user', content: userInput })
    setUserInput('')
    
    // 生成下一个问题
    setIsTyping(true)
    await generateNextQuestion(inspiration, userInput)
    setIsTyping(false)
  }
  
  return (
    <View style={styles.container}>
      {/* 顶部：杯子可视化 */}
      <View style={styles.glassContainer}>
        <Glass
          type={inspiration.type}
          completion={inspiration.completion}
          size="large"
        />
        <ProgressIndicator completion={inspiration.completion} />
      </View>
      
      {/* 中部：对话区域 */}
      <ScrollView style={styles.chatContainer}>
        {messages.map((msg, idx) => (
          <ChatBubble
            key={idx}
            message={msg}
            isUser={msg.role === 'user'}
          />
        ))}
        
        {isTyping && (
          <ChatBubble
            message={{ role: 'assistant', content: '...' }}
            isUser={false}
            isTyping
          />
        )}
      </ScrollView>
      
      {/* 底部：输入区域 */}
      <View style={styles.inputContainer}>
        <TextInput
          value={userInput}
          onChangeText={setUserInput}
          placeholder="输入你的回答..."
          multiline
          style={styles.input}
        />
        <Button
          mode="contained"
          onPress={handleSend}
          disabled={!userInput.trim() || isLoading}
        >
          发送
        </Button>
      </View>
    </View>
  )
}
```

// src/screens/BrewingScreen/ChatBubble.tsx

import React from 'react'
import { View, Text } from 'react-native'
import { useTheme } from 'react-native-paper'

interface ChatBubbleProps {
  message: { role: string; content: string }
  isUser: boolean
  isTyping?: boolean
}

export const ChatBubble: React.FC<ChatBubbleProps> = ({
  message,
  isUser,
  isTyping,
}) => {
  const theme = useTheme()
  
  return (
    <View
      style={[
        styles.container,
        isUser ? styles.userContainer : styles.assistantContainer,
      ]}
    >
      {!isUser && (
        <Text style={styles.roleLabel}>🤔 调酒师</Text>
      )}
      
      <View
        style={[
          styles.bubble,
          isUser
            ? { backgroundColor: theme.colors.primary }
            : { backgroundColor: theme.colors.surface },
        ]}
      >
        <Text
          style={[
            styles.text,
            isUser ? { color: '#fff' } : { color: theme.colors.text },
          ]}
        >
          {isTyping ? '正在思考...' : message.content}
        </Text>
      </View>
    </View>
  )
}
