## 一、功能概述

Phase 3 实现以下核心功能：

1. **灵感碰撞界面** - 用户在吧台场景多选 2-3 个灵感杯进行混合
2. **碰撞动画** - 液体倒入雪克壶、摇晃、混合的动画效果
3. **配方建议卡片** - AI 生成多个创意结合方向供用户选择
4. **碰撞历史** - 记录灵感碰撞的过程和结果

### 3.2 灵感混合器

```typescript
// src/services/collision/mixer.ts

import { Inspiration } from '../../types'
import { mixColors, generateMixType } from './utils'

export class InspirationMixer {
  /**
   * 混合灵感
   */
  mix(inspirations: Inspiration[]): MixedResult {
    // 提取颜色
    const colors = inspirations.map(insp => this.getColorByType(insp.type))
    
    // 决定混合方式（随机）
    const mixType = generateMixType()
    
    // 混合颜色
    const mixedColor = mixColors(colors, mixType)
    
    // 提取关键词
    const keywords = this.extractKeywords(inspirations)
    
    // 提取情感氛围
    const moods = this.extractMoods(inspirations)
    
    return {
      mixedColor,
      mixType,
      keywords,
      moods,
      combinedContent: this.combineContent(inspirations),
    }
  }
  
  /**
   * 根据杯子类型获取颜色
   */
  private getColorByType(type: GlassType): string {
    const colorMap = {
      [GlassType.BRANDY]: '#1E3A5F',    // 钴蓝色
      [GlassType.CHAMPAGNE]: '#9B59B6', // 淡紫色
      [GlassType.WINE]: '#8B0000',       // 深红色
      [GlassType.COCKTAIL]: '#D4A017',   // 琥珀色
      [GlassType.BEAKER]: '#00FF7F',     // 荧光绿
      [GlassType.MASON]: '#FF8C00',      // 暖橙色
      [GlassType.FLASK]: '#87CEEB',      // 透明微蓝
      [GlassType.MARTINI]: '#C0C0C0',   // 银色（混合色）
    }
    return colorMap[type]
  }
  
  /**
   * 提取关键词
   */
  private extractKeywords(inspirations: Inspiration[]): string[] {
    const allKeywords: string[] = []
    inspirations.forEach(insp => {
      if (insp.structuredContent?.keywords) {
        allKeywords.push(...insp.structuredContent.keywords)
      }
    })
    return [...new Set(allKeywords)].slice(0, 10)
  }
  
  /**
   * 提取情感氛围
   */
  private extractMoods(inspirations: Inspiration[]): string[] {
    const allMoods: string[] = []
    inspirations.forEach(insp => {
      if (insp.structuredContent?.mood) {
        allMoods.push(insp.structuredContent.mood)
      }
    })
    return [...new Set(allMoods)]
  }
  
  /**
   * 合并内容
   */
  private combineContent(inspirations: Inspiration[]): string {
    return inspirations
      .map(insp => insp.rawInput?.text || '')
      .filter(Boolean)
      .join(' + ')
  }
}
```

### 3.3 配方生成器

```typescript
// src/services/collision/generator.ts

import { LLMService } from '../llm'
import { PromptTemplate } from '../llm/prompts/brewing'
import type { Recipe, MixedResult } from '../../types'

export class RecipeGenerator {
  private llmService: LLMService
  
  constructor() {
    this.llmService = new LLMService()
  }
  
  /**
   * 生成配方建议
   */
  async generateRecipes(mixedResult: MixedResult, count: number = 3): Promise<Recipe[]> {
    const prompt = this.buildPrompt(mixedResult, count)
    
    const response = await this.llmService.chat([
      { role: 'system', content: this.systemPrompt },
      { role: 'user', content: prompt },
    ])
    
    return this.parseRecipes(response, count)
  }
  
  /**
   * 构建 Prompt
   */
  private buildPrompt(mixedResult: MixedResult, count: number): string {
    return `请为以下灵感融合生成 ${count} 个创意方向：

灵感关键词：${mixedResult.keywords.join('、')}
情感氛围：${mixedResult.moods.join('、')}
原始内容：${mixedResult.combinedContent}

请生成 ${count} 个独特的创意方向，每个方向包括：
1. 标题（简短有吸引力）
2. 详细描述（1-2句话）
3. 3个关键词
4. 2-3个可能的发展方向

请用 JSON 格式返回：`
  }
  
  /**
   * 系统提示词
   */
  private systemPrompt = `你是一位创意调酒师，擅长将多个灵感融合成独特的创意方案。

你的风格：
- 善于发现不同灵感之间的联系
- 提供多元化的创意方向
- 鼓励用户探索跨界组合
- 每个方案都要有独特的视角

请用 JSON 格式返回，结构如下：
{
  "recipes": [
    {
      "id": "方案1",
      "title": "标题",
      "description": "描述",
      "keywords": ["关键词1", "关键词2", "关键词3"],
      "directions": ["方向1", "方向2"],
      "score": 85
    }
  ]
}`
  
  /**
   * 解析响应
   */
  private parseRecipes(response: string, count: number): Recipe[] {
    try {
      const jsonMatch = response.match(/\{[\s\S]*\}/)
      if (jsonMatch) {
        const data = JSON.parse(jsonMatch[0])
        return data.recipes || []
      }
    } catch {
      console.error('[RecipeGenerator] 解析失败')
    }
    
    // 返回默认方案
    return this.generateDefaultRecipes(count)
  }
  
  /**
   * 生成默认方案
   */
  private generateDefaultRecipes(count: number): Recipe[] {
    return Array.from({ length: count }, (_, i) => ({
      id: `recipe_${i + 1}`,
      title: `创意方案 ${i + 1}`,
      description: '一个独特的创意方向',
      keywords: ['创意', '灵感', '融合'],
      directions: ['深入发展', '跨界探索'],
      score: 70 + Math.random() * 20,
    }))
  }
}
```

#### 3.4.2 摇晃动画

```typescript
// src/screens/CollisionScreen/ShakeAnimation.tsx

import React, { useEffect } from 'react'
import { View, StyleSheet } from 'react-native'
import Animated, {
  useSharedValue,
  useAnimatedStyle,
  withRepeat,
  withSequence,
  withTiming,
  withDelay,
  Easing,
} from 'react-native-reanimated'
import { Text } from 'react-native-paper'

export const ShakeAnimation: React.FC<{
  colors: string[]
  onComplete: () => void
}> = ({ colors, onComplete }) => {
  const rotation = useSharedValue(0)
  const translateY = useSharedValue(0)
  const opacity = useSharedValue(1)
  
  useEffect(() => {
    // 倒入动画
    translateY.value = withDelay(
      500,
      withTiming(-100, { duration: 800, easing: Easing.inOut(Easing.eio) })
    )
    
    // 摇晃动画
    rotation.value = withDelay(
      1500,
      withRepeat(
        withSequence(
          withTiming(-15, { duration: 150 }),
          withTiming(15, { duration: 150 })
        ),
        8,
        false
      )
    )
    
    // 倒出动画
    opacity.value = withDelay(
      3000,
      withTiming(0, { duration: 800 }, () => {
        onComplete()
      })
    )
  }, [])
  
  const shakerStyle = useAnimatedStyle(() => ({
    transform: [
      { rotate: `${rotation.value}deg` },
      { translateY: translateY.value },
    ],
    opacity: opacity.value,
  }))
  
  return (
    <View style={styles.container}>
      {/* 倒入动画 */}
      <Animated.View style={[styles.pourContainer, { opacity: opacity.value }]}>
        {colors.map((color, i) => (
          <View key={i} style={[styles.pourStream, { backgroundColor: color }]} />
        ))}
      </Animated.View>
      
      {/* 雪克壶 */}
      <Animated.View style={[styles.shaker, shakerStyle]}>
        <View style={styles.shakerBody}>
          <View style={styles.shakerContent}>
            {/* 混合颜色 */}
            <View style={styles.mixedColor} />
          </View>
        </View>
      </Animated.View>
      
      <Text style={styles.text}>正在调制特调...</Text>
    </View>
  )
}
```

```
Step 1: 选中的杯子飞到屏幕中央
        ↓
```
